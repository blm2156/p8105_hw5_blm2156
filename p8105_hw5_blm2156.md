``` r
set.seed(1)

sim_regression = function(n= 30, beta0 = 2, beta1 = 0) {
 prob3 = tibble(
    x = rnorm(n, mean = 0, sd = 1), 
    y = beta0 + beta1 * x + rnorm(n, 0, sqrt (50))
  )

ls_fit = lm(y ~ x, data = prob3) %>%
  broom::tidy ()

tibble(
  beta1_hat = ls_fit[[2,2]],
  p_value = ls_fit [[2,5]]
)

}
```
